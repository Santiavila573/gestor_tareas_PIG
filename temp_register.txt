import React, { useState, useEffect } from 'react';
import { User, Mail, Lock, Loader2, ArrowRight, Sparkles, Rocket, Check, X as XIcon } from 'lucide-react';
import { authService } from '../services/auth';
import { firebaseAuthService } from '../services/firebaseAuth';
import { FIREBASE_ENABLED } from '../services/firebase';
import { Role } from '../types';

interface RegisterProps {
    onRegisterSuccess: () => void;
    onNavigateToLogin: () => void;
}

const Register: React.FC<RegisterProps> = ({ onRegisterSuccess, onNavigateToLogin }) => {
    const [name, setName] = useState('');
    const [email, setEmail] = useState('');
    const [password, setPassword] = useState('');
    const [isLoading, setIsLoading] = useState(false);
    const [error, setError] = useState('');

    // Password Security State
    const [passwordStrength, setPasswordStrength] = useState(0);
    const [requirements, setRequirements] = useState({
        length: false,
        upper: false,
        lower: false,
        number: false,
        special: false
    });

    useEffect(() => {
        const rules = {
            length: password.length >= 8,
            upper: /[A-Z]/.test(password),
            lower: /[a-z]/.test(password),
            number: /\d/.test(password),
            special: /[@$!%*?&]/.test(password)
        };
        
        setRequirements(rules);

        const passedRules = Object.values(rules).filter(Boolean).length;
        setPasswordStrength((passedRules / 5) * 100);
    }, [password]);

    const getStrengthColor = () => {
        if (passwordStrength <= 20) return 'bg-red-500';
        if (passwordStrength <= 40) return 'bg-orange-500';
        if (passwordStrength <= 60) return 'bg-yellow-500';
        if (passwordStrength <= 80) return 'bg-lime-500';
        return 'bg-green-500';
    };

    const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault();
        setIsLoading(true);
        setError('');

        try {
            if (FIREBASE_ENABLED) {
                await firebaseAuthService.register(name, email, password, Role.DEVELOPER);
            } else {
                await authService.register(name, email, password, Role.DEVELOPER);
            }
            onRegisterSuccess();
        } catch (err: any) {
            setError(err?.message || err?.toString() || 'Error al registrarse');
        } finally {
            setIsLoading(false);
        }
    };

    return (
        <div className="min-h-screen w-full flex bg-[#1e1e2d] overflow-hidden relative font-sans text-slate-100">

            {/* Background Ambience */}
            <div className="absolute inset-0 w-full h-full">
                <div className="absolute top-[-10%] left-[-10%] w-96 h-96 bg-[#7b68ee]/30 rounded-full mix-blend-multiply filter blur-3xl opacity-70 animate-blob"></div>
                <div className="absolute top-[-10%] right-[-10%] w-96 h-96 bg-purple-500/30 rounded-full mix-blend-multiply filter blur-3xl opacity-70 animate-blob animation-delay-2000"></div>
                <div className="absolute bottom-[-20%] left-[20%] w-96 h-96 bg-pink-500/30 rounded-full mix-blend-multiply filter blur-3xl opacity-70 animate-blob animation-delay-4000"></div>
                <div className="absolute inset-0 bg-[url('https://grainy-gradients.vercel.app/noise.svg')] opacity-20 brightness-100 contrast-150"></div>
            </div>

            {/* Main Container */}
            <div className="relative z-10 container mx-auto flex flex-col lg:flex-row items-center justify-center min-h-screen px-4 gap-12 lg:gap-24">

                {/* Left Side: Brand & Value Prop */}
                <div className="flex-1 text-center lg:text-left space-y-8 animate-in slide-in-from-left duration-700 fade-in">
                    <div className="inline-flex items-center gap-2 px-4 py-2 rounded-full bg-white/5 border border-white/10 backdrop-blur-md text-sm font-medium text-[#a89aff]">
                        <Rocket className="w-4 h-4" />
                        <span>Comienza tu viaje hoy</span>
                    </div>

                    <div className="space-y-4">
                        <h1 className="text-5xl lg:text-7xl font-bold tracking-tight leading-tight">
                            <span className="block text-transparent bg-clip-text bg-gradient-to-r from-white via-[#7b68ee]/50 to-[#7b68ee] drop-shadow-sm">
                                Únete a
                            </span>
                            <span className="block text-transparent bg-clip-text bg-gradient-to-r from-[#7b68ee] via-purple-400 to-pink-400 font-extrabold text-shadow-glow">
                                GestorTasks AI
                            </span>
                        </h1>
                        <p className="text-lg text-slate-400 max-w-xl mx-auto lg:mx-0 leading-relaxed">
                            Crea tu cuenta en segundos y accede a la plataforma de gestión más avanzada. Colabora, planifica y escala con el poder de la Inteligencia Artificial.
                        </p>
                    </div>

                    <div className="hidden lg:flex gap-4 pt-4 text-sm text-slate-500">
                        <div className="flex items-center gap-2">
                            <div className="w-2 h-2 rounded-full bg-[#7b68ee]"></div>
                            <span>Registro gratuito</span>
                        </div>
                        <div className="flex items-center gap-2">
                            <div className="w-2 h-2 rounded-full bg-purple-500"></div>
                            <span>Sin tarjeta de crédito</span>
                        </div>
                        <div className="flex items-center gap-2">
                            <div className="w-2 h-2 rounded-full bg-pink-500"></div>
                            <span>Acceso instantáneo</span>
                        </div>
                    </div>
                </div>

                {/* Right Side: Register Form */}
                <div className="w-full max-w-md">
                    <div className="bg-[#2a2b36]/50 backdrop-blur-2xl border border-white/10 rounded-3xl p-8 shadow-2xl relative">

                        <div className="relative">
                            <div className="text-center mb-8">
                                <h2 className="text-2xl font-bold text-white">Crear cuenta nueva</h2>
                                <p className="text-slate-400 text-sm mt-2">Completa tus datos para comenzar</p>
                            </div>

                            {error && (
                                <div className="mb-6 p-4 bg-red-500/10 border border-red-500/20 text-red-200 rounded-xl text-sm flex items-center gap-2">
                                    <div className="w-1.5 h-1.5 rounded-full bg-red-400"></div>
                                    {error}
                                </div>
                            )}

                            <form onSubmit={handleSubmit} className="space-y-4">
                                <div className="space-y-2">
                                    <label className="text-xs font-semibold text-[#a89aff] ml-1 uppercase tracking-wide">
                                        Nombre Completo
                                    </label>
                                    <div className="relative group/input">
                                        <User className="absolute left-4 top-3.5 h-5 w-5 text-slate-500 group-focus-within/input:text-[#7b68ee] transition-colors" />
                                        <input
                                            type="text"
                                            value={name}
                                            onChange={(e) => setName(e.target.value)}
                                            className="w-full pl-12 pr-4 py-3.5 bg-black/20 border border-white/10 rounded-xl text-slate-200 placeholder:text-slate-600 focus:outline-none focus:border-[#7b68ee]/50 focus:ring-2 focus:ring-[#7b68ee]/20 transition-all"
                                            placeholder="Ej: Sofia (Product Owner)"
                                            required
                                        />
                                    </div>
                                </div>

                                <div className="space-y-2">
                                    <label className="text-xs font-semibold text-[#a89aff] ml-1 uppercase tracking-wide">
                                        Email Profesional
                                    </label>
                                    <div className="relative group/input">
                                        <Mail className="absolute left-4 top-3.5 h-5 w-5 text-slate-500 group-focus-within/input:text-[#7b68ee] transition-colors" />
                                        <input
                                            type="email"
                                            value={email}
                                            onChange={(e) => setEmail(e.target.value)}
                                            className="w-full pl-12 pr-4 py-3.5 bg-black/20 border border-white/10 rounded-xl text-slate-200 placeholder:text-slate-600 focus:outline-none focus:border-[#7b68ee]/50 focus:ring-2 focus:ring-[#7b68ee]/20 transition-all"
                                            placeholder="sofia.po@gestortasks.ai"
                                            required
                                        />
                                    </div>
                                </div>

                                <div className="space-y-2">
                                    <label className="text-xs font-semibold text-[#a89aff] ml-1 uppercase tracking-wide">
                                        Contraseña
                                    </label>
                                    <div className="relative group/input">
                                        <Lock className="absolute left-4 top-3.5 h-5 w-5 text-slate-500 group-focus-within/input:text-[#7b68ee] transition-colors" />
                                        <input
                                            type="password"
                                            value={password}
                                            onChange={(e) => setPassword(e.target.value)}
                                            className="w-full pl-12 pr-4 py-3.5 bg-black/20 border border-white/10 rounded-xl text-slate-200 placeholder:text-slate-600 focus:outline-none focus:border-[#7b68ee]/50 focus:ring-2 focus:ring-[#7b68ee]/20 transition-all"
                                            placeholder="••••••••••••"
                                            required
                                        />
                                    </div>
                                    
                                    {/* Password Strength Indicator */}
                                    <div className="mt-3 space-y-3">
                                        <div className="h-1.5 w-full bg-slate-700/50 rounded-full overflow-hidden">
                                            <div 
                                                className={`h-full transition-all duration-300 ${getStrengthColor()}`} 
                                                style={{ width: `${passwordStrength}%` }}
                                            ></div>
                                        </div>
                                        
                                        <div className="grid grid-cols-2 gap-2">
                                            <div className={`flex items-center gap-1.5 text-xs ${requirements.length ? 'text-green-400' : 'text-slate-500'}`}>
                                                {requirements.length ? <Check className="w-3 h-3" /> : <div className="w-3 h-3 rounded-full border border-slate-600"></div>}
                                                Mínimo 8 caracteres
                                            </div>
                                            <div className={`flex items-center gap-1.5 text-xs ${requirements.upper ? 'text-green-400' : 'text-slate-500'}`}>
                                                {requirements.upper ? <Check className="w-3 h-3" /> : <div className="w-3 h-3 rounded-full border border-slate-600"></div>}
                                                Mayúscula
                                            </div>
                                            <div className={`flex items-center gap-1.5 text-xs ${requirements.lower ? 'text-green-400' : 'text-slate-500'}`}>
                                                {requirements.lower ? <Check className="w-3 h-3" /> : <div className="w-3 h-3 rounded-full border border-slate-600"></div>}
                                                Minúscula
                                            </div>
                                            <div className={`flex items-center gap-1.5 text-xs ${requirements.number ? 'text-green-400' : 'text-slate-500'}`}>
                                                {requirements.number ? <Check className="w-3 h-3" /> : <div className="w-3 h-3 rounded-full border border-slate-600"></div>}
                                                Número
                                            </div>
                                            <div className={`flex items-center gap-1.5 text-xs ${requirements.special ? 'text-green-400' : 'text-slate-500'}`}>
                                                {requirements.special ? <Check className="w-3 h-3" /> : <div className="w-3 h-3 rounded-full border border-slate-600"></div>}
                                                Carácter especial (@$!%*?&)
                                            </div>
                                        </div>
                                    </div>
                                </div>

                                <button
                                    type="submit"
                                    disabled={isLoading}
                                    className="w-full py-4 mt-6 bg-[#7b68ee] hover:bg-[#6b58de] text-white rounded-xl font-bold shadow-lg shadow-[#7b68ee]/20 active:scale-[0.98] transition-all flex items-center justify-center gap-2 group/btn"
                                >
                                    {isLoading ? (
                                        <Loader2 className="w-5 h-5 animate-spin" />
                                    ) : (
                                        <>
                                            Registrarse Ahora
                                            <ArrowRight className="w-5 h-5 group-hover/btn:translate-x-1 transition-transform" />
                                        </>
                                    )}
                                </button>
                            </form>

                            <div className="mt-8 pt-6 border-t border-white/5 text-center">
                                <p className="text-slate-400 text-sm">
                                    ¿Ya tienes cuenta?{' '}
                                    <button
                                        onClick={onNavigateToLogin}
                                        className="text-[#7b68ee] hover:text-[#a89aff] font-semibold hover:underline transition-colors"
                                    >
                                        Inicia sesión
                                    </button>
                                </p>
                            </div>
                        </div>
                    </div>
                </div>
            </div>
        </div>
    );
};

export default Register;
